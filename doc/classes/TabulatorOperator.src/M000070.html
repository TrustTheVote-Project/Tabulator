  def opx_warn(tab)
    tab.validation_errors.each {|message| opx_print("** ERROR ** #{message}\n")}
    tab.validation_warnings.each {|message| opx_print("** WARNING ** #{message}\n")}
    tab
  rescue => e
    opx_err("Fatal error while fetching Tabulator errors/warnings")
  end
