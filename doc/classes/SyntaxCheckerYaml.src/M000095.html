  def schema_is_valid_array?(schema, trace)
    schema_valid_trace("schema_is_valid_array?", schema, trace)
    ((schema.length == 1) ?
     schema_is_valid_internal?(schema[0], trace) :
     schema_valid_error(schema, "Schema array not of length 1", trace))
  end
