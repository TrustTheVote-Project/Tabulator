  def val_warn (message1, value1 = false, message2 = false, value2 = false,
                message3 = false)
    message = "#{message1}" +
      (value1 == false ? "" : " (#{value1.to_s.gsub(/[\"\[\]]/,"")})") +
      (message2 == false ? "" : " #{message2}") +
      (value2 == false ? "" : " (#{value2.to_s})") +
      (message3 == false ? "" : " #{message3}")
    self.warnings.push(message)
    false
  end
