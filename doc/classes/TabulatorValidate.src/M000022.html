  def shouldnt(message1, value1 = "")
    message = "#{message1}" +
      (value1 == "" ? "" : " (#{value1.inspect.gsub(/[\"\[\]]/,"")})")
    print("** FATAL ERROR ** #{message}\n")
    $stdout.flush
    raise Exception # For debugging so we can see where the error came from
  end
