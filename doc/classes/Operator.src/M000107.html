  def opx_file_backup()
    File.rename(opx_file_prepend(TABULATOR_DATA_FILE),
                opx_file_prepend(TABULATOR_BACKUP_FILE))
  rescue => e
    opx_err("Fatal failure of File.rename from #{TABULATOR_DATA_FILE} " +
            "to #{TABULATOR_BACKUP_FILE}", e)
  end
