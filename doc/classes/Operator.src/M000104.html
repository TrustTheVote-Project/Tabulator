  def opc_data(tab = false)
    if (tab != false)
      opx_print_tabulator_count(tab)
    elsif (opx_empty_state?())
      opx_err("Command \"data\" ignored, Tabulator state: EMPTY")
    else
      tab = opx_instantiate_tabulator()
      opx_print_tabulator_count(tab)
      opx_dump_tabulator_data(tab)
    end
  end
