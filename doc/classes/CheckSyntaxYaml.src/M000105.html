  def schema_is_valid_trace(context, schema, trace, key = '')
    return unless trace < 0
    if (key == '')
      print "#{context}: #{schema.inspect[0 .. trace.abs]}\n"
    else
      print "#{context}: #{key.to_s} #{schema.inspect[0 .. trace.abs]}\n"
    end
  end
