  def check_syntax(schema, datum, validate = true, trace = TRACE_DEFAULT)
    self.errors = []
    if (validate && !schema_is_valid?(schema))
      check_syntax_error(-1, schema, datum, trace)
    else
      check_syntax_internal(schema, datum, 0, trace)
    end
    self.errors
  end
