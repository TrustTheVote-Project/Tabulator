  def tabulator_test_answer_vote(tab, qid, k, v)
    tab.counts_questions[qid]["answer_count_list"].each do |ac|
      if (ac["answer"] == k)
        v2 = ac["count"]
        print "Checking #{k} vote count (#{v}) for question #{qid}\n"
        assert(v == v2,
               "Question #{qid} Answer #{k} vote error, expected #{v} but got #{v2}")
      end
    end
  end
